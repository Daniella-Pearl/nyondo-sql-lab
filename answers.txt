Using `?` placeholders prevents injection by separating the **SQL command** from the **user data** before the database processes them. In a template literal, the database treats everything as one long instruction, allowing malicious input to "break out" and act as a new command. With parameterized queries, the database pre-compiles the query structure first; it then treats the user input strictly as a literal value, ensuring that even a hack like `' OR 1=1` is handled as harmless text instead of executable code.

1. Full secure.js Code
JavaScript
const Database = require('better-sqlite3');
const db = new Database('nyondo_stock.db');

function searchProductSafe(name) {
    const query = `SELECT * FROM products WHERE name LIKE ?`;
    return db.prepare(query).all(`%${name}%`);
}

function loginSafe(username, password) {
    const query = `SELECT * FROM users WHERE username = ? AND password = ?`;
    return db.prepare(query).get(username, password);
}

// Security Tests
console.log('Test 1 (Search Bypass):', searchProductSafe("' OR 1=1--"));
console.log('Test 2 (UNION Attack): ', searchProductSafe("' UNION SELECT id,username,password,role FROM users--"));
console.log('Test 3 (Admin Bypass): ', loginSafe("admin'--", 'anything'));
console.log('Test 4 (Always True):  ', loginSafe("' OR '1'='1", "' OR '1'='1"));

1. Updated secure.js with Validation
JavaScript
const Database = require('better-sqlite3');
const db = new Database('nyondo_stock.db');

function searchProductSafe(name) {
    // Validation: String check, min length 2, block < > ;
    if (typeof name !== 'string' || name.length < 2 || /[<>;]/.test(name)) {
        console.log(`REJECTED: Invalid search term "${name}"`);
        return [];
    }

    const query = `SELECT * FROM products WHERE name LIKE ?`;
    return db.prepare(query).all(`%${name}%`);
}

function loginSafe(username, password) {
    // Validation: No spaces, not empty, password min 6 chars
    if (!username || typeof username !== 'string' || username.includes(' ')) {
        console.log(`REJECTED: Invalid username format "${username}"`);
        return undefined;
    }
    if (!password || typeof password !== 'string' || password.length < 6) {
        console.log(`REJECTED: Password too short`);
        return undefined;
    }

    const query = `SELECT * FROM users WHERE username = ? AND password = ?`;
    return db.prepare(query).get(username, password);
}

// Test cases
console.log('Test 1 (search "cement"):', searchProductSafe('cement'));
console.log('Test 2 (search ""):', searchProductSafe(''));
console.log('Test 3 (search "<script>"):', searchProductSafe('<script>'));
console.log('Test 4 (login admin):', loginSafe('admin', 'admin123'));
console.log('Test 5 (login "ab"):', loginSafe('admin', 'ab'));
console.log('Test 6 (login "ad min"):', loginSafe('ad min', 'pass123'));

2. Output for each Test Case

Test 2: REJECTED: Invalid search term "" followed by []
Test 3: REJECTED: Invalid search term "<script>" followed by []

Test 6: REJECTED: Invalid username format "ad min" followed by undefined

3.Explanation
Parameterization protects the database from being manipulated, while input validation acts as a gatekeeper for the whole application. By checking data types, lengths, and illegal characters (like <> tags used in XSS attacks) before the query runs, we reduce server load and prevent malicious data from being processed at all. This "Defense in Depth" ensures that even if one security layer fails, the other remains intact.
