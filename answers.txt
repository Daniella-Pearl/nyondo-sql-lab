//TASK 4
Using `?` placeholders prevents injection by separating the **SQL command** from the **user data** before the database processes them. In a template literal, the database treats everything as one long instruction, allowing malicious input to "break out" and act as a new command. With parameterized queries, the database pre-compiles the query structure first; it then treats the user input strictly as a literal value, ensuring that even a hack like `' OR 1=1` is handled as harmless text instead of executable code.

1. Full secure.js Code
JavaScript
const Database = require('better-sqlite3');
const db = new Database('nyondo_stock.db');

function searchProductSafe(name) {
    const query = `SELECT * FROM products WHERE name LIKE ?`;
    return db.prepare(query).all(`%${name}%`);
}

function loginSafe(username, password) {
    const query = `SELECT * FROM users WHERE username = ? AND password = ?`;
    return db.prepare(query).get(username, password);
}

// Security Tests
console.log('Test 1 (Search Bypass):', searchProductSafe("' OR 1=1--"));
console.log('Test 2 (UNION Attack): ', searchProductSafe("' UNION SELECT id,username,password,role FROM users--"));
console.log('Test 3 (Admin Bypass): ', loginSafe("admin'--", 'anything'));
console.log('Test 4 (Always True):  ', loginSafe("' OR '1'='1", "' OR '1'='1"));
